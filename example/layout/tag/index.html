__yname__ = "Tag List" __author__ = "Decklin Foster" __email__ = "decklin@red-bean.com" __description__ = "List of all tags." def make(self, entries, vars): tags = {} for e in self.wisdom: for t, ts in zip(e.tags, e.tag): tags.setdefault(t, ts) vars['title'] = 'Tags - %s' % vars['blogname'] vars['body'] = '\n'.join([ '

%s

' % (ts, t) for t, ts in tags.items()]) return self.expand('page', **vars)
