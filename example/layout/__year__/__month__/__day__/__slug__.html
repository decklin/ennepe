__name__ = "Entry View" __author__ = "Decklin Foster " __description__ = "Permalinkable page for a single entry." def make(self, entries, vars): body = self.expand('entries-full', entries=entries, **vars) pagetitle = '%s - %s' % (entries[0].subject, vars['blogtitle']) return self.expand('page', layout=__name__, pagetitle=pagetitle, body=body, **vars)
